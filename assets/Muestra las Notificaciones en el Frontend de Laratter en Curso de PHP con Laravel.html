      Para usar un base_url pasado de blade al elemento vue, para los links en <template>:

      <template scope=“base_url”>
      <div class=“dropdown-menu”>
      <a :href=“base_url + ‘/’ + notification.data.follower.username” class=“dropdown-item” v-for=“notification in notifications”>
      @{{ notification.data.follower.username }} te ha seguido!
      </a>
      </div>
      </template>

      <script>
      export default{
      props: [
      ‘base_url’
      ],
      data(){
      return{
      notifications: [],
      }
      },
      mounted(){
      axios.get(this.base_url + “/api/notifications”)
      .then(response => {
      this.notifications = response.data;
      });
      }
      }
      </script>
